AB Crear0(int ne)
    	
    	
   
{
  AB raiz;

  raiz = (AB)malloc(sizeof(struct AB));
  if (raiz == NULL)
    error("Memoria Insuficiente.");
  raiz->n_etiquetas = ne;
  for (int i=1; i<=(ne+1); i++) {
    raiz->hijos[i] = NULO;
  }
  return(raiz);
}

AB Crear(int ne, int eti[])
    	
    	
   
{
  AB raiz;

  raiz = (AB)malloc(sizeof(struct AB));
  if (raiz == NULL)
    error("Memoria Insuficiente.");
  raiz->n_etiquetas = ne;
  for (int i=1; i<=(ne+1); i++) {
    raiz->hijos[i] = NULO;
  }
  for (int i=1; i<=lenght(eti[]); i++) {
    raiz->etiquetas[i] = eti[i];
  }
  return(raiz);
}
